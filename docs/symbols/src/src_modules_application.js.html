1 (function() {
  2   var global = (function() {return this;}).call(),
  3       vegas = global.vegas;
  4 
  5     /**
  6      * @namespace vegas.application
  7      * @description Functions related to the desktop Application environment
  8      */
  9     vegas.application = (function() {
 10 
 11       /**
 12        * @constant SPADE
 13        * @description the spade ascii character */
 14       vegas.SPADE = String.fromCharCode(9824);
 15 
 16       var application = {
 17       /** @lends vegas.application */
 18         init: function () {
 19 
 20           if (typeof(Titanium) == 'undefined') {
 21             return false;
 22           }
 23 
 24           this.setDefaultTitle();
 25         },
 26 
 27         setDefaultTitle: function () {
 28           this.setTitle('Veg' +  vegas.SPADE + 's');
 29         },
 30 
 31         setTitle: function (title) {
 32           Titanium.UI.getMainWindow().setTitle(title);
 33         },
 34 
 35         reload: function () {
 36           window.location = '/index.html?' + +new Date();
 37         },
 38 
 39         getFreshCopy: function () {
 40           var srcDir = Titanium.Filesystem.getFile('./src');
 41           var resourceDir = Titanium.Filesystem.getResourcesDirectory();
 42           var applicationDir = Titanium.Filesystem.getApplicationDirectory();
 43 
 44           // Delete The Resource Directory
 45           resourceDir.deleteDirectory(true);
 46 
 47           // Copy application source to resource directory
 48           srcDir.copy(resourceDir.toString());
 49 
 50         },
 51 
 52         getFreshCopyAndReload: function () {
 53           this.getFreshCopy();
 54           this.reload();
 55         },
 56 
 57         toggleMaximize: function () {
 58           if (Titanium.UI.getMainWindow().isMaximized()) {
 59             Titanium.UI.getMainWindow().unmaximize();
 60           }
 61           else {
 62             Titanium.UI.getMainWindow().maximize();
 63           }
 64         },
 65         
 66         toggleFullScreen: function () {
 67           if (Titanium.UI.getMainWindow().isFullscreen()) {
 68             Titanium.UI.getMainWindow().setFullScreen(false);
 69           }
 70           else {
 71             Titanium.UI.getMainWindow().setFullScreen(true);
 72           }
 73         },
 74 
 75         increaseTransparency: function () {
 76           var transparency = Titanium.UI.getMainWindow().getTransparency();
 77           if (transparency < 1) {
 78             transparency = transparency + .025;
 79             Titanium.UI.getMainWindow().setTransparency(transparency)
 80           }
 81         },
 82 
 83         decreaseTransparency: function () {
 84           var transparency = Titanium.UI.getMainWindow().getTransparency();
 85           if (transparency > 0) {
 86             transparency = transparency - .025;
 87             Titanium.UI.getMainWindow().setTransparency(transparency)
 88           }
 89         },
 90 
 91         quit: function () {
 92           vegas.application.exit();
 93         },
 94 
 95         exit: function () {
 96           Titanium.App.exit();
 97         }
 98 
 99       };
100 
101       vegas.init.register(application);
102 
103       return application;
104 
105     }());
106 
107   vegas.module.register('application.js');
108 
109 }());
