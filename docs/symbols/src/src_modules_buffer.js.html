1 /**
  2  * @fileOverview Contains Buffer related objects and methods
  3  */
  4 (function (global) {
  5   var vegas = global.vegas;
  6 
  7   /**
  8    * @class Buffer
  9    * @memberOf vegas
 10    *
 11    * @description Creates a buffer to be used for keeping track of text
 12    * 
 13    * There can only be one active buffer. The contains the text that is in
 14    * memory, typically the active buffer will take input from the keyboard
 15    * when the editor is in insert mode.
 16    *
 17    */
 18   vegas.Buffer = function () {
 19       vegas.utils.makeObject(this, arguments);
 20   };
 21 
 22   vegas.Buffer.prototype = {
 23 
 24     init: function () {
 25       this._value = '';
 26       this.length = this._value.length;
 27     },
 28 
 29     /**
 30      * Saves the buffer to the resource
 31      */
 32     save: function () {
 33       this.resource.save();
 34     },
 35 
 36     /**
 37      *  Removes buffer from memory (Unsafe: does not save changes)
 38      *  @returns True on success, false on failure.
 39      */
 40     destroy: function (buffer_id) {
 41       return false;
 42     },
 43 
 44     /**
 45      * Change the buffer's resource to a different location
 46      */
 47     setResource: function (resource) {
 48       this.resource = resource;
 49     },
 50 
 51     activate: function () {
 52       for (var i = 0; i < this._onActivate.length; i++) {
 53         vegas.buffers._onActivate[i](this);
 54       }
 55     },
 56 
 57     onActivate: function (callback) {
 58       vegas.buffers._onActivate.push(callback);
 59     },
 60 
 61     toString: function () {
 62       return this.getValue();
 63     },
 64 
 65     setValue: function (value) {
 66 
 67       if (typeof(value) !== 'undefined') {
 68         this._value = value;
 69       }
 70       else {
 71         this.getValue();
 72       }
 73 
 74     },
 75 
 76     append: function (string) {
 77       this._value += string;
 78     },
 79 
 80     deletePreviousCharacter: function () {
 81       var bufferValue = this._value;
 82       this._value = bufferValue.substr(0, bufferValue.length - 1);
 83     },
 84 
 85     empty: function () {
 86       this._value = '';
 87     },
 88 
 89     getValue: function () {
 90       return this._value;
 91     },
 92 
 93     /*
 94      * These methods are user facing and will be available to the command line.
 95      */
 96     api: [
 97     ]
 98 
 99   };
100 
101 
102   vegas.buffers = {
103 
104     /**
105      * A list of buffers from the specified context
106      */
107     list: function (context) {
108       context = context || 'global';
109       return bufferList;
110     },
111 
112     onActivate: function (callback) {
113       vegas.buffers._onActivate.push(callback);
114     },
115 
116     seek: function (key, context) {
117       context = context || 'global';
118 
119       return bufferList;
120     }
121 
122   };
123 
124   vegas.module.register('buffer.js');
125 
126 }(this));
