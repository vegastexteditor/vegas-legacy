116     setTextPoint: function () {
117 
118       var textX, textY;
119 
120       // The beginning of the text is where the area begins and additional
121       // pixels such as gutter / margin
122       textX = this.area.x + options.bufferTextOffsetX,
123 
124       // Since the the textBaseline is set to 'bottom' (found int paint object)
125       // we have to add the height of the font as well
126       textY = this.area.y + options.bufferFontSize + options.bufferTextOffsetY;

128       this.textPoint = {x: textX, y: textY};

140     /**
141      * Paints the CommandBar region.
142      * @param {Object} view The view that this should be painted on ( defaults
143      * to vegas.session.state.activeView)
144      */
145     paint: function (view) {
146 
147       view = view || vegas.session.state.activeView;
148 
149       var ctx = view.ctx,
150           bufferValue = this.buffer.getValue(),
151           measuredBuffer = ctx.measureText(bufferValue),
152           bufferWidth = measuredBuffer.width, // @todo: will need this for overflow.
153           bufferHeight = options.bufferFontSize,
154           textPoint = this.getTextPoint(),
155           textX = textPoint.x,
156           textY = textPoint.y; // @todo: center (currently incorrect pane height)
157 
158       // Paint the commandbar background
159       ctx.fillStyle = options.backgroundColor;
160 
161       ctx.fillRect(this.area.x, this.area.y, this.area.width, this.area.height);
162 
163       // Set the font for the command text
164       ctx.font = options.bufferFontSize + "px " + options.bufferFontFamily;
165 
166       // Paint the suggestions behind the buffer text
167       if (this.suggested && this.suggested.length > 0) {
168         var suggestion = this.suggested[this.suggestedIndex];
169         ctx.fillStyle = options.bufferSuggestFontColor;
170         ctx.fillText(suggestion, textX, textY);
171       }
172 
173       // Paint the buffer text
174       ctx.fillStyle = options.bufferFontColor;
175       ctx.fillText(bufferValue, textX, textY);
176

205     /**
206      * @event Takes in text input from the keyboard and appends it to the buffer
207      * @name: CommandBar#handleTextInput
208      * @param {string} character
209      */
210     handleTextInput: function (character) {
211 
212       var self = this,
213           outputChar = true;
214       
215       // If they pressed period, most likley an object notated string, check if
216       // we should change the context to the previous property (if its valid)
217       if (character == '.') {
218         this.switchContext();
219       }
220 
221       if (character == '=') {
222 
223         var bufferWithNoSpace = utils.trim(self.buffer.toString());
224         var objectInfo = utils.getObjectInfoFromString(bufferWithNoSpace);
225 
226         if (objectInfo !== false) {
227 
228           var adjustable = {
229             outputChar: outputChar
230           };
231 
232           vegas.hook.invoke.call(this, 'variableAssignment', {
233             buffer: this.buffer,
234             cursor: this.cursor,
235             objectInfo: objectInfo
236           }, adjustable);
237 
238           outputChar = adjustable.outputChar;
239 
240         }
241 
242       }
243 
244       if (outputChar) {
245         self.buffer.append(character);
246       }
247 
248       this.updateSuggestions();
249

252     /**
253      * Switches the CommandBar.context to the object specified by the string
254      *
255      * @param {string} string Containing the dot notated location of object to
256      * switch the CommandBar.context to.
257      */
258     switchContext: function (string) {
259       // and in between the period that they pressed and the period before
260       // that (or the begining of the string is a valid object.
261 
262       // because string = string || defaultValue will return default if string is empty.
263       if (string === undefined) {
264         string = this.buffer.toString();
265       }
266 
267       var object = utils.getObjectPropertyFromString(string);
268 
269       if (object !== false && (string == '' || typeof object === 'object')) {
270         // an object was found from the string
271         var newContext;
272 
273         if (string == '') { // if its an empty string we are in the global context
274           newContext = vegas;
275         }
276         else {
277           newContext = object;
278         }
279 
280         if (newContext !== this.context) {
281 
282           var adjustable = {
283             newContext: newContext
284           };
285 
286           vegas.hook.invoke.call(this, 'variableAssignment', {
287             currentContext: this.context,
288           }, adjustable);
289 
290           this.context = newContext;
291 
292           return true;
293         }
294         else {
295           return false;
296         }
297 
298       }
299       else {
300         this.context = false;
301         return false;
302       }
303

306     /**
307      * Deletes the previous character in the commandbar handling other ramifications
308      * such as the suggestions.
309      */
310     deletePreviousCharacter: function () {
311 
312       var buffer = this.buffer.toString();
313       var lastChar = buffer.substr(-1);
314 
315       this.buffer.deletePreviousCharacter();
316 
317       buffer = this.buffer.toString();
318 
319       if (lastChar == '.' || buffer == '' || !this.context) {
320 
321         buffer = this.buffer.toString();
322 
323         if (buffer.indexOf('.') !== -1) {
324           buffer = buffer.substring(0, buffer.lastIndexOf('.'));
325         }
326         else {
327           buffer = '';
328         }
329 
330         this.switchContext(buffer);
331       }
332 
333       this.updateSuggestions();
334       this.paint();

352     /**
353      * Updates the CommandBar.suggested to the approriate suggestions given
354      * the current buffer or the optional bufferValue paramater.
355      */
356     updateSuggestions: function (bufferValue) {
357 
358       bufferValue = bufferValue || this.buffer.toString();
359 
360       // No suggestions needed if the buffer is empty.
361       if (bufferValue == '') {
362         this.suggested = [];
363         return;
364       }
365 
366       this.suggested = [];
367 
368       var contextStrings = bufferValue.split('.');
369 
370       var contextString = contextStrings[contextStrings.length - 1];
371 
372       var suggestion_prefix;
373 
374       if (bufferValue.lastIndexOf('.') !== -1) {
375       suggestion_prefix = bufferValue.substring(0, bufferValue.lastIndexOf('.')) + '.';
376       }
377       else{
378         suggestion_prefix = '';
379       }
380 
381       for (var key in this.context) {
382         var value = this.context[key];
383         if ((key[0] >= 'a') && (key[0] <= 'z')) {
384           if (key.substring(0, contextString.length) == contextString) {
385             this.suggested.push(suggestion_prefix + key);
386           }
387         }
388       }
389

392     /**
393      * Given an object notated string, get suggestions from the objects properties
394      * 
395      * @param {string} suggestion An object notated string
396      *
397      * @depreciated
398      */
399     useSuggestion: function (suggestion) {
400 
401 //      debugger;
402 
403       // Get the primary suggestion (if not specified)
404       suggestion = suggestion || this.suggested[this.suggestedIndex];
405 
406       if (typeof suggestion !== "string") {
407         return false;
408       }
409 
410       var suggestionObjects = suggestion.split('.'),
411           suggestionKey= suggestionObjects[suggestionObjects.length - 1];
412 
413       // If the suggestion is an object
414       if (this.context[suggestionKey]) {
415 
416         // Use the suggestion
417         this.buffer.setValue(suggestion);
418 
419         // Change the current context to the suggestions scope
420         this.switchContext()
421 
422         this.updateSuggestions();
423 
424         this.paint();
425 
426       }

455     isAssigningInternalProperty: function (string) {
456 
457       var stringParts = string.split('='),
458           isAssignment = false;
459 
460       if (stringParts.length === 2) {
461 
462           var objectNotatedString = utils.trim(stringParts[0]),
463               objectNotationTokens = objectNotatedString.split('.'),
464               objectProperty = objectNotationTokens.pop(),
465               objectString = objectNotationTokens.join('.'),
466               object = utils.getObjectPropertyFromString(objectString),
467               value = utils.trim(stringParts[1]);
468 
469           if (object && objectProperty in object) {
470 
471             var matchRawValue = value.match(/^["'](.*)["'];?$/);
472 
473             if (matchRawValue) {
474               isAssignment = true;
475             }
476 
477           }
478 
479       }
480 
481       return isAssignment;
482 
483     },

485     assignInternalProperty: function (string) {
486 
487       var stringParts = string.split('=');
488 
489       var objectNotatedString = utils.trim(stringParts[0]),
490           objectNotationTokens = objectNotatedString.split('.'),
491           objectProperty = objectNotationTokens.pop(),
492           objectString = objectNotationTokens.join('.'),
493           object = utils.getObjectPropertyFromString(objectString),
494           value = utils.trim(stringParts[1]);
495 
496         var matchRawValue = value.match(/^["'](.*)["'];?$/);
497 
498         if (matchRawValue) {
499           value = utils.trim(matchRawValue[1]);
500           object[objectProperty] = value;
501         }

505     /**
506      * Executes a given command
507      * 
508      * @param {string} command A string containing the command to execute.
509      * @example
510      *  doCommand('file.save');
511      */
512     doCommand: function (cmd) {
513 
514       cmd = cmd || this.buffer.getValue();
515 
516       var actionTokens = cmd.split('.'),
517           action = vegas,
518           actionQuestionMark;
519 
520       for (var i = 0; i < actionTokens.length; i++) {
521 
522         actionQuestionMark = action[actionTokens[i]];
523 
524         if (typeof(actionQuestionMark) !== 'undefined') {
525 
526           if (typeof(actionQuestionMark) == 'function') {
527             actionQuestionMark.call(action);
528             this.buffer.empty();
529           }
530           else if (typeof(actionQuestionMark) == 'object') {
531             action = actionQuestionMark;
532             this.buffer.empty();
533             this.resetSuggestions();
534           }
535         }
536         else {
537 
538            // ... but we have valid suggestions ready
539           if (this.suggested.length > 0) {
540             // Instead of erroring out, execute the currently active suggestion
541             this.doCommand(this.suggested[this.suggestedIndex]);
542           }
543           else { // There are no suggestions... so their just totally typing random stuff?
544 
545             if (this.isAssigningInternalProperty(cmd)) {
546               this.assignInternalProperty(cmd);
547             }
548             else if (this.isCallingInternalFunction(cmd)) {
549               this.callInternalFunction(cmd);
550             }
551             else {
552               if (options.allowEval) {
553                 this.evaluate(cmd);
554               }
555               else {
556                 console.error('Command not found');
557               }
558             }
559 
560           }
561 
562         }
563 
564       }
565 
566       vegas.paint.paint();
567 
568     },

584     /**
585      * @namespace vegas.command
586      * @description api for commands are executed with(vegas) commands are just
587      * execution of methods in the vegas object.
588      *
589      * .buffer.seek('test'); = vegas.buffer.seek('test');
590      *
591      * This is a good way to navigate the application source.
592      *
593      * However for day to day, these commands can expand from reading the editor
594      * source code.
595      * 
596      * @example
597      *  .b[tab] expands to .buffer[property_list_display] (vegas.buffer)
598      *
599      *  .b[tab]s expands to .buffer.seek('[active]'); (vegas.buffer.seek('x'))
600      *
601      *  .b[tab]s[tab]test expands to vegas.buffer.seek('test');
602      *
603      */
604     vegas.command = (function(){
605 
606       var command = {
607         /** @lends vegas.command */
608         init: function () {
609 
610           var self = this;
611 
612             self.attachEvents();
613 
614         },
615 
616         attachEvents: function () {
617           
618           utils.onReady(function () {
619 
620             utils.takeTextInput(function (character) {
621               var activeRegion = vegas.region.getActiveRegion();
622               if (activeRegion.entity == 'CommandBar') {
623                   activeRegion.handleTextInput.call(activeRegion, character);
624                   activeRegion.paint();
625               }
626             });
627 
628           });
629 
630         },
631 
632         paint: function (view) {
633 
634           view = view || vegas.session.state.activeView;
635 
636           var i = view.regionList.length;
637 
638           while (i--) {
639             if (view.regionList[i].entity == 'CommandBar') {
640               view.regionList[i].paint.call(view.regionList[i]);
641             }
642           }
643 
644         },
645 
646         api: [
647           this.exec
648         ]
649 
650       };
651 
652       vegas.init.register(command);
653 
654       return command;
655 
656     }());
