 74     /**
 75      * @eventhandler: keydown
 76      *
 77      * Takes in a key event (keydown, keypress, keyup) objects and normalizes
 78      * these events into human readable keys via the getKeyName method, once
 79      * nomalized it uses this information along with the keyMapping settings
 80      * to associate associate they key events to the proper commands and
 81      * executes them once a key combo or key sequence has bee matched.
 82      */
 83     keydownHandler: function (e){
 84 
 85       var self = this,
 86           code = e.keyCode,
 87           keyName = this.getKeyName(code),
 88           keysDown,
 89           keyMapping;
 90 
 91       this.isKeyDown = true;
 92 
 93       // Only capture the key down once, Dont let the key repeat its self
 94       // (chrome will fire a bunch of events while its pressed down)
 95       if (typeof this.keysDown[code] == 'undefined') {
 96 
 97         // object property assignment tracking what keys were pressed
 98         this.keysDown[code] = keyName;
 99 
100         // Adding to an array to keep the order of what was assigned
101         this.keysDownOrder.push(code);
102 
103         // Get an array of keys down (in order that they were pressed)
104         keysDown = this.getKeysDown();
105 
106         // See if the keys that are currently down are matched to somthing
107         keyMapping = this.areKeysMapped(keysDown);
108 
109         // A shortcut was found!!!
110         if (keyMapping !== false) {
111           e.preventDefault();
112 
113           // Execute the action of the key mapping
114           this.execute(keyMapping);
115    
116           // Remove the last key pressed so that a user can press it again
117           // to repeat the action
118           delete this.keysDown[code]; // Remove from the object
119           this.keysDownOrder.pop(); // Remove from the array
120 
121         }
122 
123       }
124     },

139         // A shortcut was found!!!
140         if (keyMapping !== false) {

143             // Execute the action of the key mapping

156

158         delete this.isKeyDown;
159       }
160

163     keyupHandler: function (e){
164       e.preventDefault();
165 
166       var code = e.keyCode,
167           keysDownOrder = this.keysDownOrder,
168           keysDownOrderLen = keysDownOrder.length,
169           i = 0;
170 
171       for (; i < keysDownOrderLen; i++) {
172         if (keysDownOrder[i] == code) {
173           this.keysDownOrder.splice(i, 1);
174           break;
175         }
176       }
177 
178       if (this.isKeyDown) {
179         delete this.isKeyDown;
180       }
181 
182       delete this.keysDown[code];

186     /**
187      * Finds the literal function of object notated string.
188      *
189      * @example:
190      *  given the actionString "select.toBeginningOfDocument" it will find the
191      *  vegas.select.toBeginningOfDocument function, if it doesn't find the
192      *  function then it will return false and log an error.
193      * 
194      */
195     execute: function (keyMapping) {
196 
197       var objectProperties = keyMapping.action.split('.'), // Split the string's dot notation
198           object = vegas,
199           propertyInQuestion,
200           property;
201 
202       // Loop through the strings object notation
203       for (var i = 0; i < objectProperties.length; i++) {
204 
205         property = objectProperties[i];
206 
207         propertyInQuestion = object[property];
208 
209         // The property actually exists in code
210         if (property in object) {
211 
212           // It's a function, so execute it
213           if (typeof(propertyInQuestion) == 'function') {
214             // make sure its executed in the objects context, so it doesn't
215             // screw up the functions 'this'
216             propertyInQuestion.call(object);
217             this.executed = keyMapping;
218             return true;
219           }
220           // if its a property, change the context of the loop to it so we can
221           // continue searching through the object
222           else if (typeof(propertyInQuestion) == 'object') {
223             object = propertyInQuestion;
224           }
225         }
226         else {
227 
228           var activeRegion = vegas.session.state.activeRegion
229 
230           if (activeRegion.entity == keyMapping.context && property in activeRegion && typeof(activeRegion[property]) == 'function') {
231              activeRegion[property]();
232             this.executed = keyMapping;
233             return true;
234           }
235           else {
236             console.error('Unidentified property: ', property, ' in ', object);
237             return false;
238           }
239 
240         }
241 
242       }

285     /**
286      * Creates quick hash table for associating keyCodes to keyNames, used by
287      * the getKeyCode Method
288      */
289     keyNameToKeyCode: function (type) {
290 
291       type = type || 'keydown';
292 
293       var keyboard = vegas.settings.keyboard[type],
294           keyCode,
295           keyName,
296           keyNameKeyCodeMapping = this.keyNameKeyCodeMapping[type];
297 
298           keyNameKeyCodeMapping[type] = {};
299 
300       for (keyCode in keyboard) {
301         keyName = keyboard[keyCode];
302         keyNameKeyCodeMapping[keyName] = keyCode;
303       }
304 
305       this.keyNameKeyCodeMapping[type] = keyNameKeyCodeMapping[type];

309     /**
310      * Since the keyHandler functions are keeping track of what keys are down
311      * manually, this function gets an optionally human readable combo string
312      *
313      * @returns an string when the friendly argument is true (ctrl+space) or
314      * an array containing the keys ['ctrl', 'space'] @todo: verify
315      *
316      */
317     getKeysDown: function (friendly) {
318       friendly = friendly || false;
319       var keyComboString = '',
320           keyComboArray = [],
321           j = 0,
322           keysDownOrderLen = this.keysDownOrder.length,
323           keyCode,
324           keysDownOrder = this.keysDownOrder,
325           keysDown = this.keysDown;
326 
327       for (; j < keysDownOrderLen; j++) {
328         keyCode = keysDownOrder[j];
329         if (friendly) {
330           keyComboString +=  '+' + keysDown[keyCode];
331         }
332         else{
333           keyComboArray.push(keysDown[keyCode]);
334         }
335       }
336 
337       if (friendly) {
338         return keyComboString.substr(1);
339       }
340       else{
341         return keyComboArray;
342       }

345     /**
346      * Checks if keys are mapped in the options
347      *
348      * From the keys array, check to see if there is a keyboard assignment for
349      * the keys, if so it will return the keyMapping object, otherwise
350      * it will return false.
351      */
352     areKeysMapped: function (keysArray) {
353 
354       var keyMappings = vegas.keyMappings.parsedKeyMappings,
355           keyMappingsLen = keyMappings.length,
356           keymapping,
357           keySequence,
358           sequenceMatches = this.sequenceMatches,
359           sequenceMatchesLen,
360           newSequenceMatches = [],
361           foundKeyMapping = false;
362 
363           // // console.log('this.sequenceIndex',this.sequenceIndex);
364 
365           // // console.log('sequenceMatches',sequenceMatches);
366 
367       // We have already matched atleast one sequence key
368       if (this.sequenceIndex > 0) {
369 
370         // // console.log('matched somthing in the sequence');
371 
372         sequenceMatchesLen = sequenceMatches.length;
373 
374         // Loop through the possible sequence matches (based on sequences they last key(s) pressed matched.
375         while (sequenceMatchesLen--) {
376           keymapping = sequenceMatches[sequenceMatchesLen];
377           keySequence = keymapping.keySequence;
378 
379           if  (typeof(keySequence[this.sequenceIndex]) !== 'undefined') {
380 
381             var matchedKey = this.keysMatch(keysArray, keySequence[this.sequenceIndex]);
382 
383             if (matchedKey !== false) {
384 
385               if (this.sequenceIndex == keySequence.length - 1) {
386                 this.sequenceMatches = [];
387                 this.sequenceIndex = 0;
388                 foundKeyMapping = keymapping;
389                 return foundKeyMapping;
390               }
391 
392               newSequenceMatches.push(keymapping);
393 
394             }
395 
396           }
397         }
398 
399         if (newSequenceMatches.length > 0) {
400           this.sequenceMatches = newSequenceMatches;
401           this.sequenceIndex++;
402         }
403         else {
404           this.sequenceMatches = [];
405           this.sequenceIndex = 0;
406         }
407 
408       }
409       else {
410         // Loop through keymappings
411         while (keyMappingsLen--){
412           keymapping = keyMappings[keyMappingsLen];
413           keySequence = keymapping.keySequence;
414 
415           // NON key sequence actions (not currently in a key sequence
416           // and we are searching through keymappings for actions
417           // that have singular key sequences.
418           if (keySequence.length == 1 && this.inSequence == false) {
419             if (this.keysMatch(keysArray, keySequence[0])) {
420 
421               // The keys seem to match, now ensure its in the correct Region assigned
422 
423               if (keymapping.context) {
424                 // The keymapping is contextual
425 
426                 var activeRegion = vegas.region.getActiveRegion();
427 
428                 if (activeRegion.entity == keymapping.context) {
429                   // The keymapping was in the correct context
430                   foundKeyMapping = keymapping;
431                   break;
432                 }
433                 else {
434                   // The keymapping was contextual, but was not in the correct context.
435                   continue;
436                 }
437 
438               }
439               else {
440                 // The keymapping is not contextual
441                 foundKeyMapping = keymapping;
442                 break;
443               }
444 
445             }
446           }
447           // match sequence keys
448           else if (keySequence.length > 1) {
449             // @todo: match a regular expression
450 
451             if  (this.keysMatch(keysArray, keySequence[0])) {
452               this.sequenceIndex++;
453               this.sequenceMatches.push(keymapping);
454             }
455 
456           }
457 
458         }
459       }
460 
461       return foundKeyMapping;
462

465     /**
466      * Checks to see if two key combinations match, this needs to be a function
467      * for two reasons, one being for the flexablility of allowing sequential
468      *  non sequential keys to be optional. The second being that some times
469      * we may want to use place holder for key types (any number)
470      *
471      * @param keys1 @todo:document
472      *
473      * @param keys2 @todo:document
474      *
475      * @param sequentially A Boolean When true, will make sure that they keys where pressed
476      * in the correct order e.g. ctrl+1+2 and ctrl+2+1 could be different
477      * if sequentially was true
478      */
479     keysMatch: function (keys1, keys2, sequentially) {
480 
481       sequentially = sequentially || false;
482 
483       var matches;
484 
485       if (typeof(keys2) === 'string') { // @todo: all cases?
486         if (keys2 == '{number}') {
487 
488           if (utils.isArray(keys1)) {
489             if (keys1.length == 1) {
490 
491               keys1 = keys1.toString(); // tis only one key.
492 
493               var isNumber = utils.isNumber();
494 
495               if (isNumber) {
496                 this.insertingNumbers = true;
497                 return false; // okay it did match, but didn't match the entire sequence.
498               }
499               else {
500                 return false;
501               }
502 
503             }
504             else {
505               // @todo: This case happends very often, work out this logic
506               console.error('what the hells going on here?');
507             }
508           }
509           else {
510               console.error('does this ever even happen?');
511           }
512 
513          // // console.log(keys1,keys2);
514 
515         }
516       }
517       // @todo:document: the following code needs to be walked through with
518       // inline comments
519       if (sequentially) {
520 
521         if (utils.isArray(keys1)) {
522           keys1 = keys1.join('+');
523         }
524 
525         if (utils.isArray(keys2)) {
526           keys2 = keys2.join('+');
527         }
528 
529         return (keys1 ===  keys2);
530       }
531       else {
532 
533         if (!utils.isArray(keys1)) {
534           keys1 = keys1.split('+');
535         }
536 
537         if (!utils.isArray(keys2)) {
538           keys2 = keys2.split('+');
539         }
540 
541         matches = 0;
542 
543         if (keys1.length !== keys2.length) {
544           return false;
545         }
546         else {
547 
548           for (var i = 0; i < keys1.length; i++) {
549             var key1 = keys1[i];
550             for (var j = 0; j < keys2.length; j++) {
551               var key2 = keys2[j];
552               if (key1 === key2) {
553                 matches++;
554               }
555             }
556 
557           }
558 
559           if (matches !== keys2.length) {
560             return false;
561           }
562           else {
563             return true;
564           }
565 
566         }
567 
568       }
569 
570     }
