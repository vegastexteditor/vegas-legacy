1 (function() {
  2   var global = (function(){return this;}).call(),
  3       vegas = global.vegas;
  4 
  5     /**
  6      * @namespace vegas.keyMappings
  7      */
  8     vegas.keyMappings = (function(){
  9 
 10       var keyMappings = {
 11         /** @lends vegas.keyMappings */
 12         keySequences: {},
 13 
 14         init: function () {
 15 
 16           // parse special syntax keymapping syntax from the keyMappings option
 17           this.parsedKeyMappings = this.parseKeyMappings();
 18 
 19           // this.attachKeyMappings();
 20 
 21         },
 22 
 23         /**
 24          *
 25          * Because generally it seems like a silly idea to define keymappings
 26          * within the modules them selves, this allows for consolidation of
 27          * key mapping related to key sequences / functions.
 28          *
 29          * It is hoped to make keymaping easier.
 30          *
 31          * There is pluses and minuses for making up this custom syntax,
 32          * however it should be easy to use. A more javascript oriented method
 33          * of keymappings should be supported.
 34          *
 35          * @todo: escape sequences, and validation / making sure the string makes
 36          * sense should be built into this function.
 37          *
 38          * Theoretical Example of keyMappings array:
 39          *  [
 40          *     'return in commandBar does doCommand',
 41          *     'ctrl+; does command.showCommandBar',
 42          *     'ctrl+a | does pane.splitVertical',
 43          *     'ctrl+tab does tab.next',
 44          *     'v e g a s in editArea while in editMode does easterEgg',
 45          *     'd /^[0-9]+$/ l in editArea while in commandMode does editArea.deleteLine(count)'
 46          *   ]
 47          * 
 48          * Desired output:
 49          *
 50          *   [{
 51          *       keySequence: ["return"],
 52          *       action: "command.doCommand",
 53          *       context: "commandBar",
 54          *       mode: false,
 55          *   },
 56          *   {
 57          *       keySequence: ["ctrl+;"],
 58          *       action: "command.showCommandBar",
 59          *       context: false,
 60          *       mode: false,
 61          *   },
 62          *   {
 63          *       keySequence: ["ctrl+a", "|"],
 64          *       action: "pane.splitVertical",
 65          *       context: false,
 66          *       mode: false,
 67          *   },
 68          *   {
 69          *       keySequence: ["ctrl+tab"],
 70          *       action: "tab.next",
 71          *       context: false,
 72          *       mode: false,
 73          *   },
 74          *   {
 75          *       keySequence: ["v", "e", "g", "a", "s"],
 76          *       action: "easterEgg",
 77          *       context: "editArea",
 78          *       mode: "editMode",
 79          *   },
 80          *   {
 81          *       keySequence: ["d", "/^[0-9]+$/", "l"],
 82          *       action: "editArea.deleteLine(count)",
 83          *       context: "editArea",
 84          *       mode: "commandMode",
 85          *   }]
 86          *
 87          */
 88         parseKeyMappings: function (keyMappings) {
 89 
 90           keyMappings = keyMappings || vegas.settings.keyMappings;
 91 
 92           var context,
 93               action,
 94               mode,
 95               inPos,
 96               doesPos,
 97               whileInPos,
 98               optionsPos,
 99               keyMappingString,
100               tokens,
101               foundKeywords,
102               firstKeywordPos,
103               keySequence,
104               token,
105               optionsMatch,
106               optionsSplit,
107               optionsLen,
108               option,
109               optionsKeyValue,
110               options = {},
111               parsedKeyMappings = [],
112               i = keyMappings.length,
113               j;
114 
115           while (i--) {
116 
117               keyMappingString = keyMappings[i];
118               foundKeywords = [];
119               firstKeywordPos = false;
120               action = false;
121               context = false;
122               mode = false;
123               keySequence = false;
124               inPos = false;
125               doesPos = false;
126               whileInPos = false;
127               optionsPos = false;
128 
129               // Split into space tokens to make things easier to work with
130               tokens = keyMappingString.split(' ');
131 
132               j = tokens.length;
133 
134               // The "options" token supplies per sequence / key detection options
135               optionsMatch = keyMappingString.match(/\((a-zA-Z0-9= ,)\)/);
136               if (optionsMatch && optionsMatch.length > 1) {
137                 optionsPos = 1; // ittle do.. no it wont.
138               }
139 
140               // Find the position in the string of each keyword
141               while (j--) {
142 
143                 token = tokens[j];
144 
145                 // The "while in" keyword determines the mode context commandMode | editMode, etc
146                 if (token == 'while') { // This keyword is split up into two tokens
147                   if (tokens[j + 1] == 'in') {
148                     whileInPos = j + 1;
149                     continue;
150                   }
151                 }
152 
153                 // The "in" keyword determines the focus context (which pane has focus)
154                 if (token == 'in') {
155                   inPos = j;
156                   continue;
157                 }
158 
159                 // The "does" keyword supplies the action to perform
160                 if (token == 'does') {
161                   doesPos = j;
162                   continue;
163                 }
164 
165               }
166 
167               // If we found the "in" keyword
168               if (inPos !== false){
169                 // the context is always provided after the first "in" keyword
170                 context = tokens[inPos + 1];
171                 // keep track of token positions
172                 foundKeywords.push(inPos);
173               }
174 
175               // If we found the "does" keywork
176               if (doesPos !== false) {
177                 // the action is always after the "does" keyword
178                 action = tokens[doesPos + 1];
179                 action = this.getTrueActionFromContext(action, context);
180                 // keep track of token positions
181                 foundKeywords.push(doesPos);
182               }
183 
184               // If we found the "while in" keyword
185               if (whileInPos !== false) {
186                 // the action is always after the "while in" keyword
187                 mode = tokens[whileInPos + 1];
188                 // keep track of token positions
189                 foundKeywords.push(whileInPos);
190               }
191 
192               if (optionsPos !== false) {
193 
194                 optionsSplit = optionsMatch[1].split(/,\s*/);
195                 optionsLen = optionsSplit.length;
196 
197                 while (optionsLen--) {
198                   option = optionsSplit[optionsLen];
199                   optionsKeyValue = option.split(/\s*\=\s*/);
200 
201                   if (optionsKeyValue.length == 1) {
202                     options[optionsKeyValue[0]] = true;
203                   }
204                   else if (optionsKeyValue.length == 2){
205                     options[optionsKeyValue[0]] = optionsKeyValue[1];
206                   }
207 
208                 }
209 
210                 foundKeywords.push(optionsPos);
211               }
212 
213               // If we found keywords
214               if (foundKeywords.length > 0) {
215                 // Get the position of the first token we found
216                 firstKeywordPos = Math.min.apply(Math, foundKeywords);
217               }
218 
219               if (firstKeywordPos) {
220                 // The key sequence is will be spaced characters before the first keyword
221                 keySequence = tokens.splice(0, firstKeywordPos);
222 
223               }
224 
225               parsedKeyMappings[i] = {
226                 keySequence: keySequence,
227                 action: action,
228                 context: context,
229                 mode: mode,
230                 options: options,
231                 keyMappingString: keyMappingString
232               };
233 
234           }
235 
236           return parsedKeyMappings;
237 
238         },
239 
240         /**
241          * Since Actions can have shorthand eg.
242          *   instead of:  return in commandBar does command.doCommand
243          *   you can use: return in commandBar does doCommand
244          *
245          * Since commandBar is a region and is obviously related to commands it
246          * has a 'commandContext' object of 'command' and everythings always
247          * going to start with vegas.
248          *
249          * After finishing writing this the concept seems pretty stupid :D
250          * 
251          */
252         getTrueActionFromContext: function (action, context) {
253 
254           var trueAction = action,
255               actionObjects = action.split('.'),
256               commandContext = false;
257 
258           // Get the commandContext if it exists
259           if (context in vegas && typeof(vegas[context].prototype) == 'object') {
260             if ('commandContext' in vegas[context].prototype) {
261               commandContext = vegas[context].prototype.commandContext;
262             }
263           }
264 
265           // Remove the vegas. prefix if added for some reason
266           if (actionObjects[0] == 'vegas') {
267             trueAction = trueAction.substr(6); // 6 = 'vegas'.length
268           }
269 
270           // Add The commandContext prefix
271           if (commandContext && actionObjects[0] !== commandContext) {
272               trueAction = commandContext + '.' + trueAction;
273           }
274 
275           return trueAction;
276 
277         },
278 
279 
280         actions: [
281           'parseKeyMappings'
282         ]
283 
284       };
285 
286       vegas.init.register(keyMappings);
287 
288       return keyMappings;
289
