155     /**
156      *
157      * Simple object initiator, for consistency, brevity and why not some saftey.
158      *
159      * http://ejohn.org/blog/simple-class-instantiation/
160      *
161      * @param self {object} "this" object from the original function
162      * @param args {object} "arguments" variable from the original function
163      * @param inheritFrom {object} "inheritFrom" an instantiatiable object to inherit from
164      *
165      * @example
166      *  var objectSample = function () {
167      *    vegas.makeObject(this, arguments);
168      *  };
169      *
170      *  objectSample.prototype = {
171      *    init: function () { // Optional
172      *    }
173      *  };
174      *
175      */
176     makeObject: function (self, args, inheritFrom) {
177 
178       // Paranoid and unnecessary error handling
179       if (typeof self !== 'object') {
180         vegas.error('bad param: self');
181       }
182 
183       if (typeof args.callee !== 'function') {
184         vegas.error('bad param: args');
185       }
186 
187       // Check to see if the object has been instantiated
188       if (self instanceof args.callee) {
189 
190         // If there is an init function, call it in its own context
191         if (typeof self.init === "function") {
192           self.init.apply(self, args);
193         }
194 
195         // get all properties of previous object.
196         if (typeof(inheritFrom) === 'function') {
197           vegas.utils.extend(self, new inheritFrom());
198         }
199 
200         return true;
201       }
202       else {
203         // Don't automatically instantiate, warn and quit.
204         vegas.error('This method must be instantiated via the new operator.');
205         return false;
206       }

240     /**
241      * Given a string and a baseObject (defaults to the primary vegas object)
242      * return the object property of a given dot notated string.
243      *
244      * This is used alot in the command module.
245      *
246      * For Example:
247      *   utils.getObjectPropertyFromString('settings.CommandBar.buffer.fontFamily');
248      *
249      *   returns: "Arial"
250      *
251      *   utils.getObjectPropertyFromString('settings.CommandBar');
252      *
253      *   returns: the command bar object.
254      *
255      *   returns false and logs a warning if an object property was not found.
256      *
257      */
258     getObjectPropertyFromString: function (objectNotatedString, baseObject) {
259 
260       var namespace = vegas.namespace;
261 
262       // The object notated string is the namespace so return the namespaced global
263       if (objectNotatedString === namespace) {
264         return vegas;
265       }
266 
267       // If the object notated string contains the namespace, remove it since
268       // we are already in it.
269       if (objectNotatedString.indexOf(namespace) === 0) {
270         objectNotatedString = objectNotatedString.substr(namespace.length + 1);
271       }
272 
273       var objectPropertyTokens = objectNotatedString.split('.'),
274           objectPropertyQuestionMark,
275           objectProperty = baseObject || vegas;
276 
277       if (objectNotatedString == '') {
278         return false;
279       }
280 
281       for (var i = 0; i < objectPropertyTokens.length; i++) {
282 
283         objectPropertyQuestionMark = objectProperty[objectPropertyTokens[i]];
284 
285         if (typeof(objectPropertyQuestionMark) !== 'undefined') {
286 
287           if (typeof(objectPropertyQuestionMark) == 'object') {
288             objectProperty = objectPropertyQuestionMark;
289           }
290           else {
291             objectProperty = objectPropertyQuestionMark;
292             break;
293           }
294 
295         }
296         else {
297           return false;
298         }
299 
300       }
301 
302       return objectProperty;
303 
304     },

306     /**
307      * Get a nifty looking unique identifier
308      */
309     getUniqueId: function () {
310      var chars = CHARS, uuid = new Array(36), rnd = 0, r;
311 
312       for (var i = 0; i < 36; i++) {
313 
314         if (i == 8 || i == 13 || i== 18 || i == 23) {
315           uuid[i] = '_';
316         }
317         else if (i == 14) {
318           uuid[i] = '4';
319         }
320         else {
321           if (rnd <= 0x02) {
322             rnd = 0x2000000 + (Math.random()*0x1000000)|0;
323           }
324           r = rnd & 0xf;
325           rnd = rnd >> 4;
326           uuid[i] = chars[(i == 19) ? (r & 0x3) | 0x8 : r];
327         }
328       }
329       return uuid.join('');
