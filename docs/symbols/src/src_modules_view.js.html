1 (function () {
  2   var global = (function () {return this;}).call(),
  3       vegas = global.vegas,
  4       session = vegas.session,
  5       utils = vegas.utils;
  6 
  7     /**
  8      * @class View
  9      * @memberOf vegas
 10      * @description  A group of Components inside of a single panes, can be visualized in (but
 11      * is not limited to) the traditional pane form.
 12      */
 13     vegas.View = (function (data) {
 14 
 15       var View = function () {
 16         utils.makeObject(this, arguments);
 17         this.entity = 'View';
 18       };
 19 
 20       View.prototype = {
 21         /** @lends vegas.View */
 22         init: function (data) {
 23           // Take in properties passed in to data object.
 24           utils.extend(this, data);
 25           this.setupWindow();
 26           this.setupViewPort();
 27         },
 28 
 29         setupWindow: function () {
 30           // Assign a unique Id to the window so we can do easy comparison of
 31           // windows, especially for finding the View object of a window object.
 32           this.context.name = utils.getUniqueId();
 33         },
 34 
 35         setupViewPort: function () {
 36           // Remove margins, and apply necessary styles to the body
 37           var bodyStyle = global.document.body.style;
 38           bodyStyle.margin = '0';
 39           bodyStyle.padding = '0';
 40           // Prevent any flicker or in weird cases a gap between canvas and body
 41           bodyStyle.backgroundColor = vegas.settings.backgroundColor;
 42 
 43           try {
 44             this.canvas = this.context.document.getElementsByTagName('canvas')[0];
 45             this.ctx = this.canvas.getContext('2d');
 46           }
 47           catch (e) {
 48             console.error('could not get the canvas object.');
 49             vegas.terminate();
 50             return false;
 51           }
 52 
 53           // Apply necessary styles to the canvas
 54 
 55           // For now just removes the scrollbar
 56           this.canvas.style.position = 'fixed';
 57 
 58           // Prevent any flicker?
 59           this.canvas.style.backgroundColor = vegas.settings.backgroundColor;
 60 
 61           // Stretch the canvas across the body
 62           vegas.view.setArea(this);
 63 
 64         },
 65 
 66         remove: function () {
 67 
 68         },
 69 
 70         moveTo: function (position) {
 71 
 72         },
 73 
 74         /*
 75          * These methods are user facing and will be available to the command line.
 76          */
 77         api: [
 78           this.create,
 79           this.remove,
 80           this.moveTo,
 81           this.getWindow
 82         ]
 83 
 84       };
 85 
 86       return View;
 87 
 88     }());
 89     /**
 90      * @namespace vegas.view
 91      */
 92     vegas.view = (function () {
 93 
 94       var view = {
 95         /** @lends vegas.view */
 96         panes: [],
 97 
 98         init: function () {
 99           this.attachEvents();
100         },
101 
102         attachEvents: function () {
103 
104           var self = this;
105           var window = self.getCurrentWindow();
106           
107           utils.onResize(window, function (e) {
108             view = self.getViewFromWindow(window); // @TODO: this doesn't need to be ran all the time.
109             self.setArea(view);
110             vegas.region.reflow(view);
111             vegas.paint.paint();
112           });
113 
114         },
115 
116         getActiveView: function () {
117           
118         },
119 
120         /**
121          * Because there is no easy connect between a View in the session
122          * and an arbitrary window.
123          */
124         getViewFromWindow: function (window) {
125           var views = vegas.session.state.views;
126           for (var i = 0; i < views.length; i++) {
127             if (views[i].context.name == window.name) {
128               return views[i];
129             }
130           }
131           return false;
132         },
133 
134         /**
135          * Stretches the canvas to take up the entire window.
136          */
137         setArea: function (view) {
138           var width = this.getViewWidth(view),
139               height = this.getViewHeight(view),
140               offset = this.getOffset(view);
141 
142           view.width = width;
143           view.height = height;
144           view.canvas.width = width;
145           view.canvas.height = height;
146           view.offset = offset;
147 
148         },
149 
150         /**
151          * Get the inner width of the window.
152          */
153         getViewWidth: function (view) {
154           return view.context.window.innerWidth;
155         },
156 
157         /**
158          * Get the inner height of the window.
159          */
160         getViewHeight: function (view) {
161           return view.context.window.innerHeight;
162         },
163 
164         getOffset: function (view) {
165 
166           var offset = vegas.utils.offset(view.canvas);
167 
168           var offsetXy = {
169             x: offset.left,
170             y: offset.top
171           };
172 
173           return offsetXy;
174         },
175 
176         /**
177          * Gets the current active window.
178          */
179         getCurrentWindow: function () {
180           return global; // @TODO:fake
181         },
182 
183         open: function () {
184           
185         },
186 
187         close: function (view) {
188           view = view || session.activeView;
189           view.close();
190         },
191 
192         api: [
193           this.open,
194           this.close
195         ]
196 
197       };
198 
199       vegas.init.register(view);
200 
201       return view;
202 
203     }());
